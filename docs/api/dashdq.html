def run_checks(config: dict, spark=None) -> DQReport: 
 94def run_checks(config: dict, spark=None) -> DQReport:
 95    """
 96    Execute all checks defined in config and return a DQReport.
 97
 98    config shape::
 99
100        {
101            "source":   {"table": "catalog.schema.table"},
102            "metadata": {"data_owner": "", "data_steward": "", ...},  # optional
103            "checks": [
104                {"check_name": "expect_column_values_to_not_be_null",
105                 "column": "customer_id",
106                 "threshold_pct": 100.0,
107                 "params": {}},
108                ...
109            ],
110            "output":   {"type": "delta", "delta_table": "..."}       # optional
111        }
112    """
113    if not config:
114        raise ValueError("Config is empty — run dashdq.configure() first and click 'Save Config'.")
115
116    if spark is None:
117        from pyspark.sql import SparkSession
118        spark = SparkSession.getActiveSession()
119
120    table = config["source"]["table"]
121    df = spark.table(table)
122    total_cols = len(df.columns)
123    metadata = config.get("metadata", {})
124    run_ts = datetime.now().isoformat(timespec="seconds")
125    tags_str = ",".join(metadata.get("tags", []))
126
127    checked_cols: set[str] = set()
128    results: list[CheckResult] = []
129
130    for chk in config.get("checks", []):
131        name = chk["check_name"]
132        col = chk.get("column", "_TABLE_LEVEL_")
133        threshold = float(chk.get("threshold_pct", 100.0))
134        params = chk.get("params", {})
135
136        entry = CHECKS_REGISTRY.get(name)
137        if not entry:
138            continue
139
140        try:
141            if entry.get("cross_table"):
142                total, passed, failed = entry["fn"](df, col, params, spark)
143            else:
144                total, passed, failed = entry["fn"](df, col, params)
145            passed_pct = round(passed / total * 100, 2) if total > 0 else 0.0
146            status = "PASS" if passed_pct >= threshold else "FAIL"
147            if not entry.get("table_level") and not entry.get("compound"):
148                checked_cols.add(col)
149        except Exception as exc:
150            total = passed = failed = 0
151            passed_pct = 0.0
152            status = f"ERROR: {exc}"
153
154        results.append(CheckResult(
155            table_name=table,
156            column_name=col,
157            check_name=name,
158            dq_dimension=entry["dimension"],
159            total_rows=total,
160            passed_rows=passed,
161            failed_rows=failed,
162            passed_pct=passed_pct,
163            threshold_pct=threshold,
164            status=status,
165            check_params=json.dumps(params),
166            run_timestamp=run_ts,
167            data_owner=metadata.get("data_owner", ""),
168            data_steward=metadata.get("data_steward", ""),
169            business_domain=metadata.get("business_domain", ""),
170            table_description=metadata.get("description", ""),
171            tags=tags_str,
172            columns_checked=0,       # back-filled below
173            total_columns=total_cols,
174            column_coverage_pct=0.0,
175        ))
176
177    # Back-fill coverage (same value for all rows — table-level metric)
178    n_covered = len(checked_cols)
179    coverage = round(n_covered / total_cols * 100, 2) if total_cols else 0.0
180    for r in results:
181        r.columns_checked = n_covered
182        r.column_coverage_pct = coverage
183
184    report = DQReport(results, config)
185
186    # Auto-save if output block present and not dataframe-only
187    output_cfg = config.get("output", {})
188    types = output_cfg.get("types") or ([output_cfg.get("type", "dataframe")])
189    if output_cfg and types != ["dataframe"]:
190        report.save(output_cfg, spark)
191
192    return report

47    def save(self, output_cfg: dict, spark=None):
48        """Persist results to one or more destinations defined in output_cfg."""
49        import os
50
51        # Support both old single-type ("type") and new multi-type ("types") format
52        types = output_cfg.get("types") or ([output_cfg["type"]] if "type" in output_cfg else ["dataframe"])
53
54        results = {}
55        sdf = None
56
57        for otype in types:
58            if otype == "dataframe":
59                if sdf is None:
60                    sdf = self.to_spark_df(spark)
61                results["dataframe"] = sdf
62
63            elif otype == "delta":
64                if sdf is None:
65                    sdf = self.to_spark_df(spark)
66                table = output_cfg.get("delta_table", "")
67                if not table:
68                    print("⚠️  delta_table not set — skipping Delta output")
69                    continue
70                (sdf.write.format("delta")
71                    .mode("append")
72                    .option("mergeSchema", "true")
73                    .saveAsTable(table))
74                print(f"✅ Saved to Delta table: {table}")
75                results["delta"] = sdf
76
77            elif otype in ("volume_json", "volume_csv"):
78                vol_path = output_cfg.get("volume_path", "").rstrip("/")
79                filename = output_cfg.get("filename") or f"dq_results_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
80                ext = "json" if otype == "volume_json" else "csv"
81                full = f"{vol_path}/{filename}.{ext}"
82                pdf = self.to_pandas()
83                os.makedirs(vol_path, exist_ok=True)
84                if ext == "json":
85                    pdf.to_json(full, orient="records", indent=2)
86                else:
87                    pdf.to_csv(full, index=False)
88                print(f"✅ Saved to: {full}")
89                results[otype] = full
90
91        return results.get("dataframe") or (sdf if sdf is not None else None)
