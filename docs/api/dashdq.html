def run_checks(config: dict, spark=None) -> DQReport: 
171def run_checks(config: dict, spark=None) -> DQReport:
172    """
173    Execute all checks defined in config and return a DQReport.
174
175    config shape::
176
177        {
178            "source":   {"table": "catalog.schema.table"},
179            "metadata": {"data_owner": "", "data_steward": "", ...},  # optional
180            "checks": [
181                {"check_name": "expect_column_values_to_not_be_null",
182                 "column": "customer_id",
183                 "threshold_pct": 100.0,
184                 "params": {}},
185                ...
186            ],
187            "output":   {"type": "delta", "delta_table": "..."}       # optional
188        }
189    """
190    if not config:
191        raise ValueError("Config is empty — run dashdq.configure() first and click 'Save Config'.")
192
193    if spark is None:
194        from pyspark.sql import SparkSession
195        spark = SparkSession.getActiveSession()
196
197    table = config["source"]["table"]
198    df = spark.table(table)
199    total_cols = len(df.columns)
200    metadata = config.get("metadata", {})
201    run_ts = datetime.now().isoformat(timespec="seconds")
202    tags_str = ",".join(metadata.get("tags", []))
203
204    checked_cols: set[str] = set()
205    results: list[CheckResult] = []
206
207    for chk in config.get("checks", []):
208        name = chk["check_name"]
209        col = chk.get("column", "_TABLE_LEVEL_")
210        threshold = float(chk.get("threshold_pct", 100.0))
211        params = chk.get("params", {})
212
213        entry = CHECKS_REGISTRY.get(name)
214        if not entry:
215            continue
216
217        try:
218            if entry.get("cross_table"):
219                total, passed, failed = entry["fn"](df, col, params, spark)
220            else:
221                total, passed, failed = entry["fn"](df, col, params)
222            passed_pct = round(passed / total * 100, 2) if total > 0 else 0.0
223            status = "PASS" if passed_pct >= threshold else "FAIL"
224            if not entry.get("table_level") and not entry.get("compound"):
225                checked_cols.add(col)
226        except Exception as exc:
227            total = passed = failed = 0
228            passed_pct = 0.0
229            status = f"ERROR: {exc}"
230
231        results.append(CheckResult(
232            table_name=table,
233            column_name=col,
234            check_name=name,
235            dq_dimension=entry["dimension"],
236            total_rows=total,
237            passed_rows=passed,
238            failed_rows=failed,
239            passed_pct=passed_pct,
240            threshold_pct=threshold,
241            status=status,
242            check_params=json.dumps(params),
243            run_timestamp=run_ts,
244            data_owner=metadata.get("data_owner", ""),
245            data_steward=metadata.get("data_steward", ""),
246            business_domain=metadata.get("business_domain", ""),
247            table_description=metadata.get("description", ""),
248            tags=tags_str,
249            columns_checked=0,       # back-filled below
250            total_columns=total_cols,
251            column_coverage_pct=0.0,
252        ))
253
254    # Back-fill coverage (same value for all rows — table-level metric)
255    n_covered = len(checked_cols)
256    coverage = round(n_covered / total_cols * 100, 2) if total_cols else 0.0
257    for r in results:
258        r.columns_checked = n_covered
259        r.column_coverage_pct = coverage
260
261    report = DQReport(results, config)
262
263    # Auto-save if output block present and not dataframe-only
264    output_cfg = config.get("output", {})
265    types = output_cfg.get("types") or ([output_cfg.get("type", "dataframe")])
266    if output_cfg and types != ["dataframe"]:
267        report.save(output_cfg, spark)
268
269    return report

 49    def table_summary(self) -> dict:
 50        """Single-row summary at table level.
 51
 52        clean_records = rows that passed every column check applied to them.
 53        overall_status = PASS only if all checks passed.
 54        """
 55        if not self.results:
 56            return {}
 57
 58        r0 = self.results[0]
 59        metadata = self.config.get("metadata", {})
 60        total_rows = r0.total_rows
 61        total_checks = len(self.results)
 62        passed_checks = sum(1 for r in self.results if r.status == "PASS")
 63        failed_checks = total_checks - passed_checks
 64        overall_status = "PASS" if failed_checks == 0 else "FAIL"
 65
 66        # Clean records: rows not flagged as failed by ANY check.
 67        # Each check reports failed_rows independently; we sum them as a
 68        # conservative lower-bound on dirty rows (exact intersection needs a join).
 69        total_failed_rows = sum(r.failed_rows for r in self.results)
 70        # Cap at total_rows to avoid negative clean counts when checks overlap
 71        dirty_rows = min(total_failed_rows, total_rows) if total_rows else 0
 72        clean_rows = max(0, total_rows - dirty_rows)
 73        clean_pct = round(clean_rows / total_rows * 100, 2) if total_rows else 0.0
 74
 75        return {
 76            "table_name":          r0.table_name,
 77            "overall_status":      overall_status,
 78            "total_rows":          total_rows,
 79            "clean_rows":          clean_rows,
 80            "dirty_rows":          dirty_rows,
 81            "clean_pct":           clean_pct,
 82            "total_checks":        total_checks,
 83            "passed_checks":       passed_checks,
 84            "failed_checks":       failed_checks,
 85            "columns_checked":     r0.columns_checked,
 86            "total_columns":       r0.total_columns,
 87            "column_coverage_pct": r0.column_coverage_pct,
 88            "run_timestamp":       r0.run_timestamp,
 89            "data_owner":          metadata.get("data_owner", ""),
 90            "data_steward":        metadata.get("data_steward", ""),
 91            "business_domain":     metadata.get("business_domain", ""),
 92            "description":         metadata.get("description", ""),
 93            "tags":                ",".join(metadata.get("tags", [])),
 94        }

107    def save(self, output_cfg: dict, spark=None):
108        """Persist results to one or more destinations defined in output_cfg."""
109        import os
110
111        # Support both old single-type ("type") and new multi-type ("types") format
112        types = output_cfg.get("types") or ([output_cfg["type"]] if "type" in output_cfg else ["dataframe"])
113
114        results = {}
115        sdf = None
116
117        for otype in types:
118            if otype == "dataframe":
119                if sdf is None:
120                    sdf = self.to_spark_df(spark)
121                results["dataframe"] = sdf
122
123            elif otype == "delta":
124                if sdf is None:
125                    sdf = self.to_spark_df(spark)
126                table = output_cfg.get("delta_table", "")
127                if not table:
128                    print("⚠️  delta_table not set — skipping Delta output")
129                    continue
130                (sdf.write.format("delta")
131                    .mode("append")
132                    .option("mergeSchema", "true")
133                    .saveAsTable(table))
134                print(f"✅ Saved to Delta table: {table}")
135                # Also write table-level summary to <table>_summary if configured
136                summary_table = output_cfg.get("summary_delta_table", "")
137                if summary_table:
138                    (self.to_table_summary_df(spark)
139                        .write.format("delta")
140                        .mode("append")
141                        .option("mergeSchema", "true")
142                        .saveAsTable(summary_table))
143                    print(f"✅ Saved table summary to: {summary_table}")
144                results["delta"] = sdf
145
146            elif otype in ("volume_json", "volume_csv"):
147                vol_path = output_cfg.get("volume_path", "").rstrip("/")
148                filename = output_cfg.get("filename") or f"dq_results_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
149                ext = "json" if otype == "volume_json" else "csv"
150                full = f"{vol_path}/{filename}.{ext}"
151                pdf = self.to_pandas()
152                os.makedirs(vol_path, exist_ok=True)
153                if ext == "json":
154                    pdf.to_json(full, orient="records", indent=2)
155                else:
156                    pdf.to_csv(full, index=False)
157                print(f"✅ Saved to: {full}")
158                # Also write table-level summary alongside
159                summary_file = f"{vol_path}/{filename}_summary.{ext}"
160                spdf = self.table_summary_pandas()
161                if ext == "json":
162                    spdf.to_json(summary_file, orient="records", indent=2)
163                else:
164                    spdf.to_csv(summary_file, index=False)
165                print(f"✅ Saved table summary to: {summary_file}")
166                results[otype] = full
167
168        return results.get("dataframe") or (sdf if sdf is not None else None)
